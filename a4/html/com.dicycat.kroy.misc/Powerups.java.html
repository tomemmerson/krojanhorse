package com.dicycat.kroy.misc;

import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.math.Vector2;
import com.dicycat.kroy.entities.FireTruck;

public class Powerups {
    public boolean visible;
    public Texture powerupTexture;
    public int x;
    public int y;
    public int powerupIndex;
    public boolean timerrunning;
    public float timer;

    public Powerups(Texture powerupTexture, int x, int y, int powerupIndex) {
        this.powerupTexture = powerupTexture;
        this.x = x;
        this.y = y;
        this.powerupIndex = powerupIndex;
        this.visible = true;
        this.timerrunning = false;
        this.timer = 15;
    }

    public void PowerupChooser(FireTruck firetruck) {
        this.timerrunning = true;
        
        switch(this.powerupIndex) {
            case 1:
                System.out.println("amped");
                firetruck.setPosition(new Vector2(3798, 3965));
                break;
            case 2:
                System.out.println("damage");
                firetruck.increaseFlowRate();
                break;
            case 3:
                System.out.println("health");
                firetruck.fullRepairTruck();
                break;
            case 4:
                System.out.println("shield");
                firetruck.canBeHit = false;
                break;
            case 5:
                System.out.println("water");
                firetruck.refillWater();
                break;
        }
    }

    public void UpdateTimer(FireTruck firetruck, float deltatime) {
        if (this.timerrunning) {
            this.timer -= deltatime;
        }

        if (this.timer <= 0) {
            switch(this.powerupIndex) {
                case 2:
                    firetruck.decreaseFlowRate();
                    break;
                case 4:
                    firetruck.canBeHit = true;
                    break;
            }

            timerrunning = false;
            this.timer = 15;
            this.visible = true;
        }
    }
}
